results = []

for metric in metrics:
    # Select series for prototyping
    series = df[metric]
    # Train- test split
    train, test = series[:'2021-02-04'], series['2021-02-05':]    
    try:
        with open(f'{metric.lower()}.pkl', 'rb') as f:
            model = pickle.load(f)
            predictions = model.predict(len(test))
            r2 = r2_score(test, predictions)
            rmse = mean_squared_error(test, predictions, squared=False)
            results.append((metric, train, test, predictions, rmse, r2))
    except Exception as e:
        pass
